class Stack
{
	public:
	int data;
	Stack *next;
};
Stack *start=NULL;

void push(int data)
{
	Stack *t=new Stack(),*r;
	t->data=data;
	t->next=NULL;
	if(start==NULL)
	start=t;
	else
	{
		r=start;
		start=t;
		t->next=r;
	}
}
